### Select with Button

<DocsComponent.Example
  initialCode={`() => 
      <InputGroup borderOption="left">
        <Select>
          <Select.Trigger trailingIcon={null} placeholder="Pick a fruit..." />
          <Select.Panel>
            {[
              'Apple',
              'Banana',
              'Orange',
              'Pineapple',
              'Mango',
              'Grape',
              'Watermelon'].map(fruit => (
              <Select.Item value={fruit} key={fruit} className="capitalize">
              {fruit}
              </Select.Item>
            ))}
          </Select.Panel>
        </Select>
        <Button>Submit</Button>
      </InputGroup>
    `}
  scope={{
    Input,
    Icon,
    InputGroup,
    Select,
    Button
  }}
  imports={{
    "@parssa/universal-ui": {
      named: ["Input", "InputGroup", "Select", "Button"]
    },
    "react-icons/hi": {
      default: "Icon"
    }
  }}
/>
